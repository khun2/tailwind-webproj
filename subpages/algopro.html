using namespace std;

bool checkTrees(vector& v, int& mid, int& desired) {
    int sum = 0;
    for(int x : v) {
        if(x - mid > 0) {
            sum += x-mid;
            if (sum >= desired) return true;
        }
    }
    return false;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    int n, desired;
    cin >> n >> desired;
    vectorv(n);
    for (int i = 0; i < n; i++) {
        cin >> v[i];
    }
    int l = 0, r = 2000000, mid;
    while(r-l > 1) {
        mid = (l+r) / 2;
        if (checkTrees(v, mid, desired)) l = mid;
        else r = mid; 
    }
    cout << l;
}
